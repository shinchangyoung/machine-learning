```
세 개의 포인터 변수를 선언하고 메모리 공간을 확보하여 두 수를 입력받아 두 수의 차의 절대값을 저장한 후 출력하는 프로그램을 작성하시오.
```

```py
def main():
    # 포인터처럼 사용할 변수들 (리스트로 감싸서 메모리 공간을 확보)
    num1 = [0]
    num2 = [0]
    result = [0]

    # 입력 받기
    a, b = map(int, input("두 수를 입력하세요 (예: 15 25): ").split())

    # 값 저장
    num1[0] = a
    num2[0] = b

    # 절대값 계산
    result[0] = abs(num1[0] - num2[0])

    # 결과 출력
    print("두 수의 차의 절대값:", result[0])

main()

```
